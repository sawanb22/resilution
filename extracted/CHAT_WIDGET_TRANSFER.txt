Purpose:
Provide a single text file containing all instructions and full source files required to extract the chat widget from the RockMyAuction project and paste it into another project.

Files to create in the target project (place together in one folder, e.g. "chat-widget"):
- chatStorage.js
- chat-widget.js
- chat-widget.css
- index.html (demo)

Steps to transfer:
1. Create a folder in your target project, e.g.:
   PowerShell:
     New-Item -ItemType Directory -Path "C:\path\to\your\project\chat-widget"
2. Copy or create the files below into that folder.
   PowerShell example for copying from current RockMyAuction project:
     Copy-Item "c:\Users\bhard\Desktop\rma\RockMyAuction\src\lib\chatStorage.js" "C:\path\to\your\project\chat-widget\chatStorage.js"
     Copy-Item "c:\Users\bhard\Desktop\rma\RockMyAuction\extracted\chat-widget.js" "C:\path\to\your\project\chat-widget\chat-widget.js"
     Copy-Item "c:\Users\bhard\Desktop\rma\RockMyAuction\extracted\chat-widget.css" "C:\path\to\your\project\chat-widget\chat-widget.css"
     (Adjust source paths as necessary.)

3. Include the widget in any HTML page in the target project using:
   <script type="module" src="/path/to/chat-widget/chat-widget.js"></script>
   Then add <rma-chat-widget></rma-chat-widget> where you want the widget to appear.

Notes:
- The widget uses ES modules and IndexedDB (no server required).
- If you use a bundler (Webpack/Vite/Parcel), import chatStorage and CSS per your bundler rules or inline CSS into the component.
- To change the DB name, edit DB_NAME in chatStorage.js.

(Place this as chatStorage.js)
```javascript
// filepath: c:\Users\bhard\Desktop\rma\RockMyAuction\src\lib\chatStorage.js
// IndexedDB Chat Storage Service - Persistent Chat History

const DB_NAME = 'RockMyAuction_ChatDB';
const DB_VERSION = 1;
const STORE_NAME = 'chatSessions';

class ChatStorage {
  constructor() {
    this.db = null;
  }

  // Initialize IndexedDB
  async init() {
    return new Promise((resolve, reject) => {
      const request = indexedDB.open(DB_NAME, DB_VERSION);

      request.onerror = () => reject(request.error);
      request.onsuccess = () => {
        this.db = request.result;
        resolve(this.db);
      };

      request.onupgradeneeded = (event) => {
        const db = event.target.result;
        
        // Create object store if it doesn't exist
        if (!db.objectStoreNames.contains(STORE_NAME)) {
          const store = db.createObjectStore(STORE_NAME, { keyPath: 'id' });
          store.createIndex('timestamp', 'timestamp', { unique: false });
          store.createIndex('title', 'title', { unique: false });
        }
      };
    });
  }

  // Ensure DB is initialized
  async ensureDB() {
    if (!this.db) {
      await this.init();
    }
  }

  // Create new chat session
  async createSession(title = 'New Chat') {
    await this.ensureDB();
    
    const session = {
      id: `session_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`,
      title: title,
      messages: [],
      timestamp: Date.now(),
      lastUpdated: Date.now()
    };

    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([STORE_NAME], 'readwrite');
      const store = transaction.objectStore(STORE_NAME);
      const request = store.add(session);

      request.onsuccess = () => resolve(session);
      request.onerror = () => reject(request.error);
    });
  }

  // Get all chat sessions
  async getAllSessions() {
    await this.ensureDB();

    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([STORE_NAME], 'readonly');
      const store = transaction.objectStore(STORE_NAME);
      const request = store.getAll();

      request.onsuccess = () => {
        const sessions = request.result;
        // Sort by lastUpdated, newest first
        sessions.sort((a, b) => b.lastUpdated - a.lastUpdated);
        resolve(sessions);
      };
      request.onerror = () => reject(request.error);
    });
  }

  // Get specific session
  async getSession(sessionId) {
    await this.ensureDB();

    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([STORE_NAME], 'readonly');
      const store = transaction.objectStore(STORE_NAME);
      const request = store.get(sessionId);

      request.onsuccess = () => resolve(request.result);
      request.onerror = () => reject(request.error);
    });
  }

  // Update session (add messages, update title)
  async updateSession(sessionId, updates) {
    await this.ensureDB();

    return new Promise(async (resolve, reject) => {
      const session = await this.getSession(sessionId);
      if (!session) {
        reject(new Error('Session not found'));
        return;
      }

      const updatedSession = {
        ...session,
        ...updates,
        lastUpdated: Date.now()
      };

      const transaction = this.db.transaction([STORE_NAME], 'readwrite');
      const store = transaction.objectStore(STORE_NAME);
      const request = store.put(updatedSession);

      request.onsuccess = () => resolve(updatedSession);
      request.onerror = () => reject(request.error);
    });
  }

  // Add message to session
  async addMessage(sessionId, message) {
    const session = await this.getSession(sessionId);
    if (!session) throw new Error('Session not found');

    const messages = [...session.messages, message];
    
    // Auto-generate title from first user message
    let title = session.title;
    if (messages.length === 1 && message.role === 'user') {
      title = message.text.substring(0, 30) + (message.text.length > 30 ? '...' : '');
    }

    return await this.updateSession(sessionId, { messages, title });
  }

  // Delete session
  async deleteSession(sessionId) {
    await this.ensureDB();

    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([STORE_NAME], 'readwrite');
      const store = transaction.objectStore(STORE_NAME);
      const request = store.delete(sessionId);

      request.onsuccess = () => resolve(true);
      request.onerror = () => reject(request.error);
    });
  }

  // Delete all sessions
  async clearAll() {
    await this.ensureDB();

    return new Promise((resolve, reject) => {
      const transaction = this.db.transaction([STORE_NAME], 'readwrite');
      const store = transaction.objectStore(STORE_NAME);
      const request = store.clear();

      request.onsuccess = () => resolve(true);
      request.onerror = () => reject(request.error);
    });
  }

  // Get last 5-10 messages for context
  getRecentMessages(messages, count = 10) {
    return messages.slice(-count);
  }
}

// Export singleton instance
const chatStorage = new ChatStorage();
export default chatStorage;
```

// filepath: c:\Users\bhard\Desktop\rma\RockMyAuction\extracted\chat-widget.js
// Simple web-component chat widget that uses chatStorage.js
import chatStorage from './chatStorage.js';
import './chat-widget.css';

class RmaChatWidget extends HTMLElement {
  constructor() {
    super();
    this.attachShadow({ mode: 'open' });
    this.currentSession = null;
  }

  connectedCallback() {
    this.render();
    this.init();
  }

  async init() {
    await chatStorage.init();
    await this.loadSessions();
  }

  render() {
    this.shadowRoot.innerHTML = `
      <style>
        /* minimal reset to allow CSS file to apply via import */
        :host { display:block; }
      </style>
      <div class="rma-container">
        <div class="rma-header">
          <strong>Chat</strong>
          <div>
            <button id="newBtn">New</button>
            <button id="clearBtn">Clear</button>
          </div>
        </div>
        <div class="rma-main">
          <div class="rma-list" id="sessionsList"></div>
          <div style="flex:1; display:flex; flex-direction:column;">
            <div class="rma-messages" id="messages"></div>
            <div class="rma-input">
              <textarea id="input" placeholder="Type a message"></textarea>
              <button id="sendBtn">Send</button>
            </div>
          </div>
        </div>
      </div>
    `;

    this.shadowRoot.getElementById('newBtn').addEventListener('click', () => this.createNew());
    this.shadowRoot.getElementById('clearBtn').addEventListener('click', () => this.clearAll());
    this.shadowRoot.getElementById('sendBtn').addEventListener('click', () => this.sendMessage());
    this.shadowRoot.getElementById('input').addEventListener('keydown', (e) => {
      if (e.key === 'Enter' && !e.shiftKey) {
        e.preventDefault();
        this.sendMessage();
      }
    });
  }

  async loadSessions() {
    const listEl = this.shadowRoot.getElementById('sessionsList');
    listEl.innerHTML = '';
    const sessions = await chatStorage.getAllSessions();
    for (const s of sessions) {
      const btn = document.createElement('button');
      btn.textContent = s.title || 'Untitled';
      btn.style.display = 'block';
      btn.style.width = '100%';
      btn.style.marginBottom = '6px';
      btn.addEventListener('click', () => this.selectSession(s.id));
      const del = document.createElement('button');
      del.textContent = '🗑';
      del.style.marginLeft = '6px';
      del.addEventListener('click', async (ev) => {
        ev.stopPropagation();
        await chatStorage.deleteSession(s.id);
        if (this.currentSession && this.currentSession.id === s.id) this.currentSession = null;
        await this.loadSessions();
        this.clearMessagesView();
      });

      const container = document.createElement('div');
      container.style.display = 'flex';
      container.style.alignItems = 'center';
      container.appendChild(btn);
      container.appendChild(del);
      listEl.appendChild(container);
    }

    // auto-create one if none exist
    if (sessions.length === 0) {
      const session = await chatStorage.createSession('Welcome');
      this.currentSession = session;
      this.renderMessages(session.messages);
      await this.loadSessions();
    }
  }

  async selectSession(sessionId) {
    const session = await chatStorage.getSession(sessionId);
    if (!session) return;
    this.currentSession = session;
    this.renderMessages(session.messages);
  }

  renderMessages(messages = []) {
    const container = this.shadowRoot.getElementById('messages');
    container.innerHTML = '';
    for (const m of messages) {
      const el = document.createElement('div');
      el.className = 'message ' + (m.role || 'assistant');
      el.textContent = m.text;
      container.appendChild(el);
    }
    container.scrollTop = container.scrollHeight;
  }

  clearMessagesView() {
    this.shadowRoot.getElementById('messages').innerHTML = '';
  }

  async createNew() {
    const session = await chatStorage.createSession('New Chat');
    await this.loadSessions();
    this.selectSession(session.id);
  }

  async clearAll() {
    if (!confirm('Clear all chats?')) return;
    await chatStorage.clearAll();
    this.currentSession = null;
    await this.loadSessions();
    this.clearMessagesView();
  }

  async sendMessage() {
    const input = this.shadowRoot.getElementById('input');
    const text = input.value.trim();
    if (!text) return;
    if (!this.currentSession) {
      this.currentSession = await chatStorage.createSession('Chat');
      await this.loadSessions();
    }

    const userMsg = { role: 'user', text, timestamp: Date.now() };
    await chatStorage.addMessage(this.currentSession.id, userMsg);
    this.currentSession = await chatStorage.getSession(this.currentSession.id);
    this.renderMessages(this.currentSession.messages);
    input.value = '';

    // optional: simple echo assistant reply (replace with API call if needed)
    setTimeout(async () => {
      const assistantMsg = { role: 'assistant', text: 'Echo: ' + text, timestamp: Date.now() };
      await chatStorage.addMessage(this.currentSession.id, assistantMsg);
      this.currentSession = await chatStorage.getSession(this.currentSession.id);
      this.renderMessages(this.currentSession.messages);
    }, 600);
  }
}

customElements.define('rma-chat-widget', RmaChatWidget);
export default RmaChatWidget;
